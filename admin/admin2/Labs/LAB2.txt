*add 2 disks
*create physical volume
	-sudo apt install ivm2
	-sudo pvcreate /dev/sdb /dev/sdc
	-sudo pvdisplay /dev/sdb /dev/sdc
*create Volume group
	-sudo vgcreate itivg /dev/sdb /dev/sdc
	-sudo vgdisplay itivg
*create two logical volume 
	-lv1 20% 
	-lv2 80% 
*create logical volume
	-sudo lvcreate -l 20%FREE -n lv1 itivg
	-sudo lvcreate -l 100%FREE -n lv2 itivg
*dusplay logical volume 
	-sudo lvdisplay /dev/itivg/lv1
	-sudo lvdisplay /dev/itivg/lv2
*create a file system and mount 
	-sudo apt-get install xfsprogs
	-sudo mkfs.xfs /dev/itivg/lv1
	-sudo mkdir/myDir
	-sudo mount /dev/itivg/lv1 /myDir
	-sudo mkfs.xfs /dev/itivg/lv2
	-sudo mkdir/mylogs
	-sudo mount /dev/itivg/lv1 /mylogs
*install apache 
	-$sudo apt install apache
	-$sudo service apache2 start
*configure a web application on apache
	-sudo nanao /etc/apache2/sites-available/-default.conf
		comment #DocumentRoot /var/www/html
		write DocumentRoot /myDir
		comment #ErrorLog ${APACHE_LOG_DIR} AND THE NEXT LINE AS WELL
		write ErrorLog /mylogs/error.log and CustomLog /mylogs/access.log combined
	-create 
		-sudo touch /mylogs/error.log  /mylogs/access.log

		-sudo nano /etc/apache2/apache2.conf
