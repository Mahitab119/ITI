1-Install network manager
	sudo apt install network-manager
	sudo nano /etc/default/grub
	edit line GRUB_CMDLINE_LINUX="", and inside "" add 
				net.ifnames=0 biosdevname=0
	sudo grub-mkconfig -o /boot/grub/grub.cfg
	sudo touch /etc/NetworkManager/conf.d/10-globally-managed-devices.conf
	sudo reboot
	
2-Configure the scenario
	sudo nmcli con add con-name iti type ethernet ifname eth0 autoconnect yes ip4 192.168.3.3
	sudo nmcli c s	
	sudo nmcli con up iti
	sudo apt install firewalld
	sudo systemctl enable --now firewalld
	sudo nmcli con mod connection.zone external
	
	in vm2:-
	sudo nmcli con add con-name itistatic type ethernet ifname eth0 autoconnect yes ip4 192.168.3.1
	sudo nmcli c s	
	sudo nmcli con up itistatic 
	sudo apt install firewalld
	sudo systemctl enable --now firewalld

3-In vm2, install apache2
	sudo apt install apache2
	
4-In vm2 enable access to apache2
	from vm1
		curl 192.168.3.1
